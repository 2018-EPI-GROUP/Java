## 二分法查找指定元素

- 从指定数组中里用二分法进行查找如果找到则输出该原粗所在的小标，如果没有则打印“没有找到！”

  ```java
  package strStr;
  
  import java.util.ArrayList;
  import java.util.Collections;
  import java.util.Scanner;
  public class way {
  
  	public static void main(String[] args) {	
  		Scanner sc = new Scanner(System.in);
  		ArrayList<Integer> array = new ArrayList<Integer>();
  		int n,i;
  		System.out.println("请输入你输入数字的个数:" );
  		n=sc.nextInt();
  		for(i=0;i<n;i++) {
  			array.add(sc.nextInt());
  		}
  		Collections.sort(array);
  		for(Integer temp:array) {
  			System.out.print(temp + " ");
  		}
  		int result = midSort(array);
  		if(result!=-1) {
  			System.out.println(result);
  		}else {
  			System.out.println("There have no numbers that you what!");
  		}
  	}
  	public static int midSort(ArrayList<Integer> matrix) {
          int left=0,right=matrix.size();//利用left和right两个数字指针指向现在最左和最右的元素
  		int mid=(left+right)/2;		//寻找中间元素
  		Scanner sc = new Scanner(System.in);
  		int a;
  		System.out.println("请输入你要查找的数:");
  		a=sc.nextInt();
  		while(left<right&&matrix.get(mid)!=a) {
  			if(a>matrix.get(mid)) {	//如果要寻找的数字大于中间元素
  				left=mid;			//更新最左边的元素
  			}else if(matrix.get(mid)>a) {//如果要寻找的数字小于中间元素
  				right=mid;			//更新最右边的元素
  			}
  			mid=(left+right)/2;		//更新最中间的元素
  		}
  		int result = 0;
  		if(matrix.get(mid)==a) {
  			result = mid;
  		}
  		else if(left>right) {
  			result = -1;
  		}
  		return result;
  	} 
  }
  ```
