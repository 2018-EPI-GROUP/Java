## 链表排序

```java
public class Demo {

	public static void main(String[] args) {
		ListNode T = creat();
		int s = 4;
		while(s-->0) {
			Scanner sc = new Scanner(System.in);
			Integer num = sc.nextInt();
			add(T,num);
		}
		print(sortList(T));
		
	}
	public static ListNode creat() {		//构建头指针
		ListNode T =new ListNode();
		T.next=null;
		return T;
	}
	public static void add(ListNode T,int num) {	//增加节点
		ListNode n = new ListNode(num);
		n.next=null;
		while(T.next!=null)
			T=T.next;
		T.next=n;
	}
	public static void print(ListNode T) {	//打印链表
		while(T!=null) {
			System.out.print(T.val + "->");
			T=T.next;
		}
	}
	public static ListNode sortList(ListNode head) {	//链表排序
		if(head==null)
			return head;
        ArrayList<Integer> ad = new ArrayList<>();
        ListNode T = head.next;
        while(T!=null) {
        	ad.add(T.val);
        	T = T.next;
        }
        System.out.println(ad);
        int num[]=new int[ad.size()];
        int i;
        for(i=0;i<ad.size();i++) {
        	num[i]=ad.get(i);
        }
        Arrays.sort(num);
        ListNode N = new ListNode(num[0]);
        for(i=1;i<num.length;i++) {
        	add(N,num[i]);
        }
        print(N);
        System.out.println();
        return N;
    }

}
```
