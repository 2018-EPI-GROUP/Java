

## 判断排序类型

- **题目描述**

  10
  2 1 5 7 3 5 1 3 6 9
  1 2 5 7 3 5 1 3 6 9

- **代码描述**

```java

public class Test {
	public static void main(String args[]) {
		final int MAX =100;
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		int []num = new int[N];
		int []sort = new int[N];
		for(int i=0;i<N;i++) {
			num[i]=sc.nextInt();
		}
		for(int i=0;i<N;i++) {
			sort[i]=sc.nextInt();
		}
		int q = judge(num,sort);
		if(q==0) Insert(sort);
		else
			sort=Merge(q, sort);
		for(int temp:sort) {
			System.out.print(temp + " ");
		}
	}
	public static int judge(int []num,int sort[]) {
		int i,count=0,j,k,flag=0;
		for(i=1;i<num.length;i++) {
			if(sort[i-1]>sort[i])
				break;
		}
		j=i;
		System.out.println(j);
		for(i=j;i<num.length;i++) {
			if(num[i]!=sort[i]) {
				System.out.println("Merge Sort!");
				return j;
			}
		}
		System.out.println("Insert Sort!");
		return 0;
    public static void Insert(int num[]) {
        int i=0,j,k,p;
        for(i=1;i<num.length;i++) {
            if(num[i-1]>num[i])
                break;
        }
        int N=i;
        if(N==num.length) return ;
        int temp=num[N];	
        for(i=N-1;i>0&&temp<num[i];i--) {
            num[i+1]=num[i];
        }
        num[i+1]=temp;
    }
    public static int [] Merge(int N,int num[]) {
        int i,k,p,j;
        int nums[]=new int[num.length];
        for(i=0;i<num.length;i++)
            nums[i]=num[i];
        for(i=0;i<num.length;i+=2*N) {
            for(j=i,k=i+N,p=i;j<num.length&&k<num.length&&j<i+N&&k<i+2*N;p++) {
                if(num[j]>num[k]) {
                    nums[p]=num[k];
                    k++;
                }else {
                    nums[p]=num[j];
                    j++;
                }
            }
            while(j<num.length&&j<i+N)
                nums[p++]=num[j++];
            while(k<num.length&&k<i+2*N)
                nums[p++]=num[k++];
        }
        return nums;
    }
}
```
